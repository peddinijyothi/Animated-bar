  <script>
    const initialHeights = Array(5).fill(100); // Array to store initial heights
    let currentHeights = [...initialHeights];
    let animationStarted = false;

    function getRandomHeight() {
      // Generate a random height between 1 and 100
      return Math.floor(Math.random() * 100) + 1;
    }

    function applyHeights() {
      const bars = document.querySelectorAll('.colorBar');
      bars.forEach((bar, index) => {
        bar.style.height = `${currentHeights[index]}%`;
      });
    }

    function changeHeights() {
      if (animationStarted) {
        currentHeights = Array.from({ length: 5 }, getRandomHeight
        applyHeights();
        setTimeout(changeHeights, 600); // Change heights every second
      }
    }

    function startAnimation() {
      if (!animationStarted) {
        animationStarted = true;
        changeHeights();
      }
    }

    function resetAnimation() {
      currentHeights = [...initialHeights];
      applyHeights();
      animationStarted = false;
    }

    function stopAnimation() {
      animationStarted = false;
    }
  </script>
